Car Parking Master: A Complete Game Development Journey from Console to SFML

🎮 Game Description:
In Car Parking Master, you control a car in a crowded parking lot. Your goal is to navigate through obstacles and park in the designated spot without collisions. The game tracks moves, time, and efficiency, rewarding players who park quickly and skillfully.

🧱 Basic Features (Text-Based Prototype)

  • Grid-based map (10x10)

  • Player controls using WASD keys

  • Obstacles (#) block movement

  • Parking spot (P) is the target

  • Car (C) represents the player

  • Score system rewards fewer moves

✅ Tools Used

  • Language: C++

  • Graphics (future scope): SFML / SDL

  • Sound effects (future scope): SFML Audio / SDL_mixer

🔣 Sample Text-Based Code (C++ Console)

cpp
 
#include <iostream>
#include <vector>
#include <conio.h>  // for _getch() on Windows
using namespace std;

const int SIZE = 10;
char grid[SIZE][SIZE];
int carX = 0, carY = 0;
int moves = 0;

void initGrid() {
    for (int i = 0; i < SIZE; i++)
        for (int j = 0; j < SIZE; j++)
            grid[i][j] = '.';

    // Place some obstacles
    grid[3][4] = '#';
    grid[4][4] = '#';
    grid[5][4] = '#';

    // Parking spot
    grid[9][9] = 'P';

    // Car start
    carX = 0;
    carY = 0;
    grid[carX][carY] = 'C';
}

void drawGrid() {
    system("cls");  // clear screen on Windows
    for (int i = 0; i < SIZE; i++) {
        for (int j = 0; j < SIZE; j++) {
            cout << grid[i][j] << ' ';
        }
        cout << endl;
    }
    cout << "Moves: " << moves << endl;
    cout << "Use W A S D to move the car." << endl;
}

bool moveCar(char dir) {
    int newX = carX, newY = carY;
    if (dir == 'w') newX--;
    else if (dir == 's') newX++;
    else if (dir == 'a') newY--;
    else if (dir == 'd') newY++;

    // Bounds check
    if (newX < 0 || newX >= SIZE || newY < 0 || newY >= SIZE)
        return false;

    // Obstacle check
    if (grid[newX][newY] == '#')
        return false;

    // Move car
    grid[carX][carY] = '.';
    carX = newX;
    carY = newY;
    grid[carX][carY] = 'C';
    moves++;

    // Win check
    if (carX == 9 && carY == 9)
        return true;

    return false;
}

int main() {
    initGrid();
    drawGrid();

    while (true) {
        char input = _getch();
        if (input == 'w' || input == 'a' || input == 's' || input == 'd') {
            bool success = moveCar(input);
            drawGrid();
            if (success) {
                cout << "🎉 You parked the car successfully in " << moves << " moves!" << endl;
                break;
            }
        }
    }
    return 0;
}

⏱️ Timer + 🎯 Scoring System Design

We’ll:

  • Track completion time.

  • Reward faster completions with higher scores.

  • Use chrono for timing.

Score Formula:
score = max(1000 - (moves * 10 + time_in_seconds * 5), 0)
(Encourages fewer moves and quick decisions.)

🧾 Full SFML C++ Code

cpp
 
#include <SFML/Graphics.hpp>
#include <SFML/Window.hpp>
#include <iostream>
#include <vector>
#include <chrono>

using namespace std;
using namespace sf;
using namespace chrono;

const int GRID_SIZE = 10;
const int TILE_SIZE = 64;
int carX = 0, carY = 0;
int moves = 0;
bool gameFinished = false;

vector<vector<char>> grid(GRID_SIZE, vector<char>(GRID_SIZE, '.'));

void initGrid() {
    // Place obstacles
    grid[3][4] = '#';
    grid[4][4] = '#';
    grid[5][4] = '#';

    // Parking spot
    grid[9][9] = 'P';

    // Car start
    grid[carX][carY] = 'C';
}

int calculateScore(int moves, int timeTakenSec) {
    int rawScore = 1000 - (moves * 10 + timeTakenSec * 5);
    return max(rawScore, 0);
}

int main() {
    RenderWindow window(VideoMode(GRID_SIZE * TILE_SIZE, GRID_SIZE * TILE_SIZE + 100), "🚗 Car Parking Master (SFML)");

    Font font;
    if (!font.loadFromFile("arial.ttf")) {
        cout << "Font loading failed!" << endl;
        return -1;
    }

    initGrid();
    RectangleShape car(Vector2f(TILE_SIZE - 8, TILE_SIZE - 8));
    car.setFillColor(Color::Blue);

    RectangleShape obstacle(Vector2f(TILE_SIZE, TILE_SIZE));
    obstacle.setFillColor(Color::Red);

    RectangleShape parking(Vector2f(TILE_SIZE, TILE_SIZE));
    parking.setFillColor(Color::Green);

    // Start the timer
    auto startTime = high_resolution_clock::now();

    while (window.isOpen()) {
        Event event;
        while (window.pollEvent(event)) {
            if (event.type == Event::Closed)
                window.close();

            if (!gameFinished && event.type == Event::KeyPressed) {
                int newX = carX, newY = carY;

                if (event.key.code == Keyboard::Up) newX--;
                else if (event.key.code == Keyboard::Down) newX++;
                else if (event.key.code == Keyboard::Left) newY--;
                else if (event.key.code == Keyboard::Right) newY++;

                if (newX >= 0 && newX < GRID_SIZE && newY >= 0 && newY < GRID_SIZE && grid[newX][newY] != '#') {
                    grid[carX][carY] = '.';
                    carX = newX;
                    carY = newY;
                    grid[carX][carY] = 'C';
                    moves++;

                    if (carX == 9 && carY == 9) {
                        gameFinished = true;
                    }
                }
            }
        }

        window.clear(Color::Black);

        // Draw grid
        for (int i = 0; i < GRID_SIZE; i++) {
            for (int j = 0; j < GRID_SIZE; j++) {
                RectangleShape tile(Vector2f(TILE_SIZE - 2, TILE_SIZE - 2));
                tile.setPosition(j * TILE_SIZE, i * TILE_SIZE);
                tile.setFillColor(Color(100, 100, 100));

                if (grid[i][j] == '#') {
                    obstacle.setPosition(j * TILE_SIZE, i * TILE_SIZE);
                    window.draw(obstacle);
                } else if (grid[i][j] == 'P') {
                    parking.setPosition(j * TILE_SIZE, i * TILE_SIZE);
                    window.draw(parking);
                }

                window.draw(tile);
            }
        }

        // Draw car
        car.setPosition(carY * TILE_SIZE + 4, carX * TILE_SIZE + 4);
        window.draw(car);

        // Show moves and score
        Text infoText;
        infoText.setFont(font);
        infoText.setCharacterSize(20);
        infoText.setFillColor(Color::White);

        if (gameFinished) {
            auto endTime = high_resolution_clock::now();
            auto duration = duration_cast<seconds>(endTime - startTime);
            int timeTaken = duration.count();
            int score = calculateScore(moves, timeTaken);

            infoText.setString("🎉 Parked in " + to_string(moves) + " moves, " +
                               to_string(timeTaken) + " sec\n🏆 Score: " + to_string(score));
        } else {
            infoText.setString("Moves: " + to_string(moves));
        }

        infoText.setPosition(10, GRID_SIZE * TILE_SIZE + 10);
        window.draw(infoText);

        window.display();
    }

    return 0;
}
