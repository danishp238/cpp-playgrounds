“Farm Frenzy: Harvest & Grow”

🎯 Game Objective

You're a farmer who must:

  • Plant crops 🌱

  • Feed animals 🐄

  • Harvest and sell products 🧺

  • Expand your farm 🌾

All while managing time, money, and resources wisely.

🧱 Core Game Features

Feature Description
🧑‍🌾 Farm Dashboard Shows money, weather, and day count
🌱 Crop Management Plant seeds, water them, wait, harvest
🐄 Animal Management Feed cows/chickens, collect milk/eggs
🛒 Market System Sell products for money
⏳ Time Progression Each turn = 1 day
💰 Resource Strategy Use money to buy more seeds/animals
 

📘 Basic Version: Console-based Game in C++

🧾 Sample Code Overview (farm_game.cpp)

cpp
#include <iostream> #include <vector> #include <string> #include <cstdlib> using namespace std; struct Crop { string name; int daysToGrow; int currentDay; int value; bool isHarvested = false; }; struct Animal { string type; int feedRequired; bool isFed = false; int productValue; }; int main() { int money = 100; int day = 1; vector<Crop> crops; vector<Animal> animals; cout << "🚜 Welcome to Farm Frenzy!\n"; while (true) { cout << "\n📅 Day " << day << " | 💰 Money: $" << money << "\n"; cout << "1. Plant Crop\n2. Water Crops\n3. Harvest Crops\n4. Buy Animal\n5. Feed Animals\n6. Sell Products\n7. End Day\n8. Exit\nChoose: "; int choice; cin >> choice; if (choice == 1) { cout << "Choose crop:\n1. Carrot ($10)\n2. Wheat ($15)\n"; int c; cin >> c; if (c == 1 && money >= 10) { crops.push_back({"Carrot", 3, 0, 25}); money -= 10; } else if (c == 2 && money >= 15) { crops.push_back({"Wheat", 4, 0, 40}); money -= 15; } else { cout << "❌ Not enough money.\n"; } } else if (choice == 2) { for (auto& crop : crops) { if (!crop.isHarvested && crop.currentDay < crop.daysToGrow) crop.currentDay++; } cout << "🪣 Crops watered.\n"; } else if (choice == 3) { for (auto& crop : crops) { if (crop.currentDay >= crop.daysToGrow && !crop.isHarvested) { crop.isHarvested = true; cout << "✅ Harvested " << crop.name << "\n"; } } } else if (choice == 4) { cout << "Choose animal:\n1. Cow ($50)\n2. Chicken ($20)\n"; int a; cin >> a; if (a == 1 && money >= 50) { animals.push_back({"Cow", 2, false, 30}); money -= 50; } else if (a == 2 && money >= 20) { animals.push_back({"Chicken", 1, false, 10}); money -= 20; } else { cout << "❌ Not enough money.\n"; } } else if (choice == 5) { for (auto& animal : animals) { animal.isFed = true; } cout << "🐄 All animals fed.\n"; } else if (choice == 6) { for (auto& crop : crops) { if (crop.isHarvested) { money += crop.value; cout << "Sold " << crop.name << " for $" << crop.value << endl; crop.isHarvested = false; } } for (auto& animal : animals) { if (animal.isFed) { money += animal.productValue; cout << "Collected from " << animal.type << ": $" << animal.productValue << endl; animal.isFed = false; } } } else if (choice == 7) { day++; } else if (choice == 8) { cout << "👋 Thanks for playing Farm Frenzy!\n"; break; } else { cout << "Invalid option.\n"; } } return 0; }
