üì¶ File: PicnicGameManager.h

#pragma once #include "CoreMinimal.h" #include "GameFramework/Actor.h" #include "PicnicGameManager.generated.h" UENUM(BlueprintType) enum class EPicnicSeason : uint8 { Spring, Summer, Winter }; UCLASS() class PICNICPANIC_API APicnicGameManager : public AActor { GENERATED_BODY() public: APicnicGameManager(); protected: virtual void BeginPlay() override; virtual void Tick(float DeltaTime) override; private: void ChangeSeason(); void SpawnPicnicItems(); void HandleWeather(); void UpdateScore(); FTimerHandle SeasonTimer; int32 TotalScore; UPROPERTY(EditAnywhere) EPicnicSeason CurrentSeason; UPROPERTY(EditAnywhere) TSubclassOf<AActor> PicnicMatClass; UPROPERTY(EditAnywhere) TSubclassOf<AActor> FoodItemClass; UPROPERTY(EditAnywhere) class USoundCue* BackgroundMusic; };

üß† Explanation (Word-by-Word)

  • UENUM(BlueprintType) defines a blueprint-accessible season type (Spring, Summer, Winter).

  • APicnicGameManager inherits from AActor, meaning it will be placed in the world and updated every tick.

  • BeginPlay() starts timers and initializes the picnic.

  • Tick() allows real-time control of weather, animal spawns, or camera behavior.

  • FTimerHandle SeasonTimer is used to trigger season changes.

  • TotalScore tracks your performance (cleanliness, food saved, time, etc).

  • TSubclassOf<AActor> allows dynamic spawning of food and mats.

  • USoundCue handles our background music.

#include "PicnicGameManager.h" #include "Kismet/GameplayStatics.h" #include "Sound/SoundCue.h" APicnicGameManager::APicnicGameManager() { PrimaryActorTick.bCanEverTick = true; CurrentSeason = EPicnicSeason::Spring; TotalScore = 0; } void APicnicGameManager::BeginPlay() { Super::BeginPlay(); SpawnPicnicItems(); GetWorld()->GetTimerManager().SetTimer(SeasonTimer, this, &APicnicGameManager::ChangeSeason, 60.0f, true); UGameplayStatics::PlaySound2D(this, BackgroundMusic); } void APicnicGameManager::Tick(float DeltaTime) { Super::Tick(DeltaTime); HandleWeather(); } void APicnicGameManager::SpawnPicnicItems() { GetWorld()->SpawnActor<AActor>(PicnicMatClass, FVector(200, 200, 100), FRotator::ZeroRotator); GetWorld()->SpawnActor<AActor>(FoodItemClass, FVector(250, 200, 100), FRotator::ZeroRotator); } void APicnicGameManager::ChangeSeason() { switch (CurrentSeason) { case EPicnicSeason::Spring: CurrentSeason = EPicnicSeason::Winter; break; case EPicnicSeason::Winter: CurrentSeason = EPicnicSeason::Summer; break; case EPicnicSeason::Summer: default: CurrentSeason = EPicnicSeason::Spring; break; } UE_LOG(LogTemp, Warning, TEXT("Season changed to: %d"), (int32)CurrentSeason); }
