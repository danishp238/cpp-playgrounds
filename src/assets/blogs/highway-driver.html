Highway Driver: A Complete SFML Racing Game with Audio and Progressive Difficulty

üéÆ Game Description:
In Highway Driver, you control a car speeding down a highway while avoiding incoming traffic. The longer you survive, the higher your score. The game features endless scrolling, collision detection, and increasing difficulty to keep the challenge fresh.

‚úÖ Game Features

  • Player controls: Move left/right with arrow keys.

  • Endless scrolling road: Creates a dynamic driving experience.

  • Random enemy cars: AI-controlled vehicles approach from the top.

  • Collision detection: Crash into an enemy, and it's game over!

  • Score system: Based on survival time.

  • Progressive difficulty: Speed and spawn rates increase over time.

  • Sound effects: Crash sounds and background music.

üßæ Full SFML Code (Highway Driver)

cpp
 
#include <SFML/Graphics.hpp>
#include <SFML/Window.hpp>
#include <SFML/Audio.hpp>
#include <vector>
#include <ctime>
#include <cstdlib>

using namespace sf;
using namespace std;

const int windowWidth = 400;
const int windowHeight = 600;
const float carSpeed = 5.0f;
const float baseEnemySpeed = 4.0f;
const float laneWidth = 100.0f;

struct Car {
    RectangleShape shape;
    float speed;

    Car(Color color, Vector2f pos) {
        shape.setSize(Vector2f(50, 100));
        shape.setFillColor(color);
        shape.setPosition(pos);
        speed = 0;
    }
};

bool checkCollision(Car& a, Car& b) {
    return a.shape.getGlobalBounds().intersects(b.shape.getGlobalBounds());
}

int main() {
    srand(time(0));
    RenderWindow window(VideoMode(windowWidth, windowHeight), "üöò Highway Driver");
    window.setFramerateLimit(60);

    // Player Car
    Car player(Color::Blue, Vector2f(windowWidth / 2 - 25, windowHeight - 120));

    // Enemy Cars
    vector<Car> enemies;
    float enemySpeed = baseEnemySpeed;
    float spawnInterval = 1.0f;
    float spawnTimer = 0;

    // Road Lines
    RectangleShape roadLine(Vector2f(10, 50));
    roadLine.setFillColor(Color::White);
    vector<float> lineOffsets = {100, 300, 500};

    // Game State
    Clock clock;
    int score = 0;
    int level = 1;
    float levelTimer = 0.0f;
    bool gameOver = false;

    // Font & Text
    Font font;
    if (!font.loadFromFile("arial.ttf")) return -1;

    Text scoreText;
    scoreText.setFont(font);
    scoreText.setCharacterSize(24);
    scoreText.setFillColor(Color::White);
    scoreText.setPosition(10, 10);

    Text levelText;
    levelText.setFont(font);
    levelText.setCharacterSize(24);
    levelText.setFillColor(Color::White);
    levelText.setPosition(10, 40);

    // Audio
    SoundBuffer crashBuffer;
    if (!crashBuffer.loadFromFile("crash.wav")) {
        cout << "Failed to load crash.wav" << endl;
        return -1;
    }
    Sound crashSound;
    crashSound.setBuffer(crashBuffer);

    Music bgMusic;
    if (!bgMusic.openFromFile("music.ogg")) {
        cout << "Failed to load music.ogg" << endl;
        return -1;
    }
    bgMusic.setLoop(true);
    bgMusic.setVolume(50);
    bgMusic.play();

    while (window.isOpen()) {
        Event event;
        while (window.pollEvent(event)) {
            if (event.type == Event::Closed)
                window.close();
        }

        if (!gameOver) {
            // Input Handling
            if (Keyboard::isKeyPressed(Keyboard::Left) 
                player.shape.move(-carSpeed, 0);
            if (Keyboard::isKeyPressed(Keyboard::Right)) 
                player.shape.move(carSpeed, 0);

            // Keep player in bounds
            float playerX = player.shape.getPosition().x;
            if (playerX < 0) player.shape.setPosition(0, player.shape.getPosition().y);
            if (playerX > windowWidth - 50) player.shape.setPosition(windowWidth - 50, player.shape.getPosition().y);

            // Time Tracking
            float deltaTime = clock.restart().asSeconds();
            spawnTimer += deltaTime;
            levelTimer += deltaTime;

            // Level Up System
            if (levelTimer >= 10.0f) {
                level++;
                levelTimer = 0;
                enemySpeed += 0.5f;
                spawnInterval = max(0.4f, spawnInterval - 0.1f);
            }

            // Enemy Spawning
            if (spawnTimer >= spawnInterval) {
                float x = (rand() % 3) * laneWidth + 25;
                enemies.push_back(Car(Color::Red, Vector2f(x, -100)));
                spawnTimer = 0;
            }

            // Move Enemies
            for (auto& enemy : enemies)
                enemy.shape.move(0, enemySpeed);

            // Collision Detection
            for (auto& enemy : enemies) {
                if (checkCollision(player, enemy)) {
                    crashSound.play();
                    gameOver = true;
                    break;
                }
            }

            // Remove Off-Screen Enemies
            enemies.erase(remove_if(enemies.begin(), enemies.end(),
                [](Car& c) { return c.shape.getPosition().y > windowHeight; }), enemies.end());

            // Update Score
            score++;
        }

        // Drawing
        window.clear(Color(30, 30, 30));

        // Draw Road Lines
        for (auto& y : lineOffsets) {
            roadLine.setPosition(windowWidth / 2 - 5, y);
            window.draw(roadLine);
            y += 5;
            if (y > windowHeight) y = -50;
        }

        // Draw Player & Enemies
        window.draw(player.shape);
        for (auto& enemy : enemies)
            window.draw(enemy.shape);

        // Draw UI
        scoreText.setString("Score: " + to_string(score));
        levelText.setString("Level: " + to_string(level));
        window.draw(scoreText);
        window.draw(levelText);

        // Game Over Screen
        if (gameOver) {
            Text gameOverText("CRASH! Game Over!", font, 32);
            gameOverText.setFillColor(Color::Yellow);
            gameOverText.setPosition(60, windowHeight / 2 - 50);
            window.draw(gameOverText);
        }

        window.display();
    }

    return 0;
}
