üßæ GUI Version: recipe_gui.cpp (Simplified Core)

Here's the simplified version with GUI using SFML. It covers:

  • Recipe display

  • Grocery item selection via buttons

  • Cart and budget display

cpp
#include <SFML/Graphics.hpp> #include <vector> #include <string> #include <sstream> #include <iostream> using namespace sf; using namespace std; struct Item { string name; float price; }; vector<Item> store = { {"Tomato", 1.5}, {"Onion", 1.0}, {"Chicken", 5.0}, {"Garlic", 1.2}, {"Oil", 2.0}, {"Spaghetti", 2.5} }; vector<string> recipeIngredients = {"Spaghetti", "Tomato", "Onion", "Garlic", "Oil"}; float budget = 10.0; vector<string> cart; float total = 0.0; bool allIngredientsCollected() { for (auto& ing : recipeIngredients) { if (find(cart.begin(), cart.end(), ing) == cart.end()) return false; } return true; } string floatToStr(float val) { stringstream ss; ss << "$" << fixed << setprecision(2) << val; return ss.str(); } int main() { RenderWindow window(VideoMode(800, 600), "Recipe Rush - GUI"); Font font; font.loadFromFile("arial.ttf"); // Make sure arial.ttf is in the same directory Text title("Recipe: Spaghetti", font, 24); title.setPosition(20, 20); Text budgetText("Budget: " + floatToStr(budget), font, 20); budgetText.setPosition(20, 60); vector<RectangleShape> buttons; vector<Text> labels; for (size_t i = 0; i < store.size(); i++) { RectangleShape btn(Vector2f(200, 40)); btn.setPosition(50, 120 + i * 50); btn.setFillColor(Color(200, 200, 255)); buttons.push_back(btn); Text label(store[i].name + " - " + floatToStr(store[i].price), font, 18); label.setPosition(60, 125 + i * 50); labels.push_back(label); } Text cartText("", font, 18); cartText.setPosition(300, 120); Text resultText("", font, 20); resultText.setPosition(300, 400); while (window.isOpen()) { Event event; while (window.pollEvent(event)) { if (event.type == Event::Closed) window.close(); else if (event.type == Event::MouseButtonPressed) { for (size_t i = 0; i < buttons.size(); ++i) { if (buttons[i].getGlobalBounds().contains(event.mouseButton.x, event.mouseButton.y)) { // Add item Item it = store[i]; if (total + it.price <= budget && find(cart.begin(), cart.end(), it.name) == cart.end()) { cart.push_back(it.name); total += it.price; resultText.setString("‚úÖ Added " + it.name); } else if (total + it.price > budget) { resultText.setString("‚ùå Over budget!"); } else { resultText.setString("‚ö†Ô∏è Already in cart."); } } } } } // Update cart display stringstream ss; ss << "üß∫ Cart:\n"; for (string& s : cart) ss << "- " << s << "\n"; ss << "\nSpent: " << floatToStr(total); cartText.setString(ss.str()); if (allIngredientsCollected()) { resultText.setString("üéâ All ingredients collected!"); } window.clear(Color::White); window.draw(title); window.draw(budgetText); for (size_t i = 0; i < buttons.size(); ++i) { window.draw(buttons[i]); window.draw(labels[i]); } window.draw(cartText); window.draw(resultText); window.display(); } return 0; }
